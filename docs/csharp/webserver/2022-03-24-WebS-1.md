## Task 써보기

```csharp
static Task Test()
{
    Console.WriteLine("* Start Test *");
    Task t = Task.Delay(3000);
    return t;
}

static void Main(string[] args)
{
    Task t = Test();

    Console.WriteLine("* Start While *");
    while(true)
    {

    }

    /*
        * Start While *     <- While이 먼저 뜬다??
        * Start Test *

        Task의 완료 여부를 기다리지 않고 다음줄로 넘어간다.
    */
}
```

```csharp
static Task Test()
{
    Console.WriteLine("* Start Test *");
    Task t = Task.Delay(3000);
    return t;
}

static void Main(string[] args)
{
    Task t = Test();
    t.Wait();           // t의 완료를 기다려 달라.

    Console.WriteLine("* Start While *");
    while(true)
    {

    }

    /*
        * Start Test *
        // 3초 후
        * Start While *
    */
}
```

## Async, Await 써보기

```csharp
static void TestAsync()
{
    Console.WriteLine("* Start TestAsync *");
    Task t = Task.Delay(3000);  // <- 복잡한 로직이라 가정한다.
    t.Wait();
    Console.WriteLine("* End TestAsync *");
}

static void Main(string[] args)
{
    TestAsync();

    Console.WriteLine("* Start While *");
    while(true)
    {

    }

    /*
        * Start TestAsync *
        // 3초후
        * End TestAsync *
        * Start While *
    */  
}
```

```csharp
static async void TestAsync()
{
    Console.WriteLine("* Start TestAsync *");
    await Task.Delay(3000);
    Console.WriteLine("* End TestAsync *");
}

static void Main(string[] args)
{
    TestAsync();

    Console.WriteLine("* Start While *");
    while(true)
    {

    }

    /*
        * Start TestAsync *
        * Start While *
        // 3초후
        * End TestAsync *
    */  
}
```

```csharp
static async Task Main(string[] args)
{
    Coffee cup = PourCoffee();
    Console.WriteLine("coffee is ready");

    var eggsTask = FryEggsAsync(2);
    var baconTask = FryBaconAsync(3);
    var toastTask = MakeToastWithButterAndJamAsync(2);

    var breakfastTasks = new List<Task> { eggsTask, baconTask, toastTask };
    while (breakfastTasks.Count > 0)
    {
        // 셋중 끝난애가 있는지 체크
        Task finishedTask = await Task.WhenAny(breakfastTasks);
        if (finishedTask == eggsTask)
        {
            Console.WriteLine("eggs are ready");
        }
        else if (finishedTask == baconTask)
        {
            Console.WriteLine("bacon is ready");
        }
        else if (finishedTask == toastTask)
        {
            Console.WriteLine("toast is ready");
        }
        breakfastTasks.Remove(finishedTask);
    }

    Juice oj = PourOJ();
    Console.WriteLine("oj is ready");
    Console.WriteLine("Breakfast is ready!");
}
```
