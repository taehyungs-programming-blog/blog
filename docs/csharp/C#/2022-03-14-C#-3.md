## class 생성

```csharp
class Knight
{
    public int hp;
    public int attack;

    public void Move()
    {
        Console.WriteLine("Knight Move");
    }

    public void Attack()
    {
        Console.WriteLine("Knight Attack");
    }
}

class Program
{
    static void Main(string[] args)
    {
        Knight knight = new Knight();

        knight.hp = 100;
        knight.Move();
        knight.Attack();
    }
}
```

## 복사

```csharp
class Knight
{
    // ...
}

struct Mage
{
    // ...
}

class Program
{
    static void KillKnight(Knight k)
    {
        // 들어온 캐릭을 죽인다.
        k.hp = 0;
    }

    static void KillMage(Mage k)
    {
        // 들어온 캐릭을 죽인다.
        k.hp = 0;
    }

    static void Main(string[] args)
    {
        Knight knight1 = new Knight();
        knight1.hp = 100;

        Mage Mage2;
        Mage2.hp = 100;

        KillKnight(knight1);      // 죽음
        KillMage(Mage2);          // hp = 100 그대로 임.

        /*
            class - call by reference
            struct - call by value
        */

        Mage Mage21 = Mage2;
        // 둘은 동일한 객체가 아니다(복사가 일어난다.)
        Knight knight11 = knight1;
        // 둘은 동일한 객체이다 (참조로 동작한다)
    }
}
```

```csharp
class Knight
```

## 상속성

```csharp
class Player
{
    static public int counter = 0;
    public int hp;
    public int attack;

    public Player()
    {
    }

    public Player(int hp)
    {
        this.hp = hp;
    }
}

class Mage : Player
{
}

class Knight : Player
{
    public Knight(int hp) : Player(hp)
    {
        base.attack = 100;  // 부모의 변수에 접근 가능
    }
}
```

```csharp
class Program
{
    static void EnterGame(Player p)
    {
        bool isMage = (player is Mage);
        if (isMage)
        {
            Mage mage = (Mage)p;
            p.mp = 100;
        }

        // or

        Mage mage = (player as Mage);
        if (mage != null)
        {
            p.mp = 100;
        }
    }

    static void Main(string[] args)
    {
        Knight knight = new Knight();
        Mage mage = new Mage();

        EnterGame(knight);
        EnterGame(mage);
```
