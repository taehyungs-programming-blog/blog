## K - 1213 *

* [ë¬¸ì œ ğŸŒ](https://www.acmicpc.net/problem/1213)

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int freq[26]; // to keep the frequency of each character
char ans[51]; // to keep the final palindrome

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    string s; cin >> s;
    for (char c : s) freq[c - 'A']++;

    int cnt = 0;
    for (int i = 0; i < 26; i++) if (freq[i] % 2) cnt++; // count the odd frequency

    if (cnt > 1) {
        cout << "I'm Sorry Hansoo";
        return 0;
    }

    int front = 0, back = s.size() - 1;
    for (int i = 0; i < 26; i++) {
        for (int j = 0; j < freq[i] / 2; j++) {
            ans[front++] = ans[back--] = i + 'A';
        }
    }

    for (int i = 0; i < 26; i++) {
        if (freq[i] % 2) {
            ans[s.size() / 2] = i + 'A';
            break;
        }
    }

    for (int i = 0; i < s.size(); i++) cout << ans[i];

    return 0;
}
```

## L - 1940 *

* [ë¬¸ì œ ğŸŒ](https://www.acmicpc.net/problem/1940)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    cin >> N;
    cin >> M;

    vector<int> materials(N);

    for (int i = 0; i < N; ++i) {
        cin >> materials[i];
    }

    sort(materials.begin(), materials.end());

    int count = 0;
    int start = 0;
    int end = N - 1;

    while (start < end) {
        if (materials[start] + materials[end] == M) {
            ++count;
            ++start;
        }
        else if (materials[start] + materials[end] < M) {
            ++start;
        }
        else {
            --end;
        }
    }

    cout << count << "\n";

    return 0;
}
```
