## 예제 H - 13913

* [문제 🌎](https://www.acmicpc.net/problem/13913)

```cpp
#include <iostream>
#include <queue>
#include <vector>

using namespace std;

const int MAX = 100000;
int dist[MAX + 1];  // 수빈이의 위치부터의 최단 시간을 저장하는 배열
int from[MAX + 1];  // 어디서부터 이동했는지 저장하는 배열

void bfs(int start, int end) {
    queue<int> q;
    q.push(start);
    dist[start] = 0;

    while (!q.empty()) {
        int current = q.front();
        q.pop();

        if (current == end)
            break;

        // 걷는 경우: X-1, X+1
        vector<int> walk = {current - 1, current + 1};
        // 순간이동하는 경우: 2*X
        int teleport = 2 * current;

        for (int next : walk) {
            if (next >= 0 && next <= MAX && dist[next] == -1) {
                q.push(next);
                dist[next] = dist[current] + 1;
                from[next] = current;
            }
        }

        if (teleport >= 0 && teleport <= MAX && dist[teleport] == -1) {
            q.push(teleport);
            dist[teleport] = dist[current] + 1;
            from[teleport] = current;
        }
    }
}

void printPath(int start, int end) {
    if (start != end)
        printPath(start, from[end]);

    cout << end << ' ';
}

int main() {
    int N, K;
    cin >> N >> K;

    fill(dist, dist + MAX + 1, -1);  // dist 배열 초기화
    fill(from, from + MAX + 1, -1);  // from 배열 초기화

    bfs(N, K);

    cout << dist[K] << '\n';
    printPath(N, K);

    return 0;
}
```

## 예제 M - 2529

* [문제 🌎](https://www.acmicpc.net/problem/2529)

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

bool check(vector<char>& signs, vector<int>& nums) {
    for (int i = 0; i < signs.size(); i++) {
        if (signs[i] == '<' && nums[i] > nums[i + 1])
            return false;
        if (signs[i] == '>' && nums[i] < nums[i + 1])
            return false;
    }
    return true;
}

int main() {
    int k;
    cin >> k;

    vector<char> signs(k);
    for (int i = 0; i < k; i++) {
        cin >> signs[i];
    }

    vector<int> maxNums(k + 1);
    vector<int> minNums(k + 1);
    for (int i = 0; i <= k; i++) {
        maxNums[i] = 9 - i;
        minNums[i] = i;
    }

    while (true) {
        if (check(signs, maxNums))
            break;
        prev_permutation(maxNums.begin(), maxNums.end());
    }

    while (true) {
        if (check(signs, minNums))
            break;
        next_permutation(minNums.begin(), minNums.end());
    }

    for (int i = 0; i <= k; i++) {
        cout << maxNums[i];
    }
    cout << endl;

    for (int i = 0; i <= k; i++) {
        cout << minNums[i];
    }
    cout << endl;

    return 0;
}
```

## 예제 L - 1987

* [문제 🌎](https://www.acmicpc.net/problem/1987)

```cpp
#include <bits/stdc++.h>
using namespace std;

int R, C, ret, ny, nx, visited[30];
char a[21][21];
const int dx[] = {-1, 0, 1, 0};
const int dy[] = {0, 1, 0, -1};

void go(int y, int x, int cnt) {
    ret = max(ret, cnt);
    for(int i = 0; i < 4; i++) {
        ny = y + dy[i];
        nx = x + dx[i];
        if(ny < 0 || ny >= R || nx < 0 || nx >= C) continue;
        int next = (int)(a[ny][nx] - 'A');

        if(Visited[next] == 0) {
            visited[next] = 1;
            go(ny, nx, cnt + 1);
            visited[next] = 0;
        }
    }

    return ;
}

int main()
{
    cin >> R >> C;
    for(int i = 0; i < R; i++) {
        for(int j = 0; j < C; j++) {
            cin >> a[i][j];
        }
    }

    visited[(int)a[0][0] = 'A']  = 1;
    go(0, 0, 1);
    cout << ret << '\n';
    return 0;
}
```
