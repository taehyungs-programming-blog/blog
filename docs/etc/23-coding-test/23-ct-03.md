void bfs(int start, int end) {

    q.push(start);
    dist[start] = 0;

    while (!q.empty()) {
        int current = q.front();
        q.pop();

        if (current == end)
            break;

        // ê±·ëŠ” ê²½ìš°: X-1, X+1
        vector<int> walk = {current - 1, current + 1};
        // ìˆœê°„ì´ë™í•˜ëŠ” ê²½ìš°: 2*X

        for (int next : walk) {
            if (next >= 0 && next <= MAX && dist[next] == -1) {
                q.push(next);
                dist[next] = dist[current] + 1;
                from[next] = current;
            }
        }

        if (teleport >= 0 && teleport <= MAX && dist[teleport] == -1) {
            q.push(teleport);
            dist[teleport] = dist[current] + 1;
            from[teleport] = current;
        }
    }
}

    if (start != end)
        printPath(start, from[end]);

    cout << end << ' ';

#include <iostream>
#include <vector>
#include <queue>
#include <cmath>
#include <cstring>
using namespace std;

int N, L, R;
int A[51][51];
bool visited[51][51];
int dx[4] = { -1, 0, 1, 0 };
int dy[4] = { 0, 1, 0, -1 };
vector<pair<int, int>> unions;
int total_people = 0;

void BFS(int x, int y) {
    queue<pair<int, int>> q;
    q.push({ x,y });
    visited[x][y] = true;
    unions.push_back({ x,y });
    total_people += A[x][y];

    while(!q.empty()) {
        int cx = q.front().first;
        int cy = q.front().second;
        q.pop();

        for(int i = 0; i < 4; i++) {
            int nx = cx + dx[i];
            int ny = cy + dy[i];
            if(nx >= 0 && nx < N && ny >= 0 && ny < N && !visited[nx][ny]) {
                int diff = abs(A[cx][cy] - A[nx][ny]);
                if(diff >= L && diff <= R) {
                    q.push({ nx, ny });
                    visited[nx][ny] = true;
                    unions.push_back({nx, ny});
                    total_people += A[nx][ny];
                }
            }
        }
    }
}

int main() {
    cin >> N >> L >> R;
    for(int i = 0; i < N; i++) {
        for(int j = 0; j < N; j++) {
            cin >> A[i][j];
        }
    }

    int result = 0;
    while(true) {
        memset(visited, false, sizeof(visited));
        bool moved = false;
        for(int i = 0;i < N; i++) {
            if(!visited[i][j]) {
                unions.clear();
                total_people = 0;
                BFS(i, j);
                if(unions.size() >= 2) {
                    moved = true;
                    int average_people = total_people / unions.size();
                    for(int k = 0; k < unions.size(); k++) {
                        A[unions[k].first][unions[k].second] = average_people;
                    }
                }
            }
        }
        if(!moved) break;
        result++;
    }
    cout << result << endl;
    return 0;
}
