#include <algorithm>
#include <string>
#include <vector>
#include <map>

using namespace std;
map<string, int> combi;

void combination(string src, string crs, int depth) {
    if (crs.size() == depth) combi[crs]++;

    else for (int i = 0; i < src.size(); i++)
        combination(src.substr(i + 1), crs + src[i], depth);
}

vector<string> solution(vector<string> orders, vector<int> course) {
    vector<string> answer;

    for (string& order : orders)
        sort(order.begin(), order.end());

    for (int crs : course) {
        for (string order : orders)
            // ëª¨ë“  ê¸€ìì˜ ì½¤ë¹„ë„¤ì´ì…˜ë³„ ì¹´ìš´íŠ¸ë¥¼ ë„£ì–´ì¤€ë‹¤

        for (auto it : combi)
            // ìµœëŒ€ ëª‡ê°œì¸ì§€ ì²´í¬í•´ì£¼ê³

    }

    sort(answer.begin(), answer.end());
    return answer;
}
