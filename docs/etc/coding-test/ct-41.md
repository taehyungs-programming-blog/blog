#include <string>
#include <vector>

using namespace std;

#include <algorithm>

bool comp(int a, int b) {
	return a < b;
}

int solution(int k, int m, vector<int> score) {
	
	//sort(score.begin(), score.end(), greater<int>());
	sort(score.begin(), score.end());
	
	vector<vector<int>> s;

	while (true)
	{
		if (score.size() < m) break;
		vector<int> v;
		for (int i = 0; i < m; i++)
		{
			v.push_back(score[score.size()-1]);
			score.pop_back();
		}
		s.push_back(v);
	}

	int answer = 0;
	for (auto i : s)
	{
		if (i.empty()) break;
		sort(i.begin(), i.end(), comp);
		answer += m * i[0];
	}

	return answer;
}

int main()
{
	solution(3, 4, {  });
	return 0;
	/*
	3	4	[1, 2, 3, 1, 2, 3, 1]	8
	4	3	[4, 1, 2, 2, 4, 4, 4, 4, 1, 2, 4, 2]	33
	*/
}
