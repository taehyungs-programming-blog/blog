#include <string>
#include <vector>
#include <map>
#include <sstream>
#include <iostream>

using namespace std;

vector<int> solution(vector<int> fees, vector<string> records) {
	
	map<string, int> m; // <car Number, minute>
	vector<int> answer;

	for (auto i : records)
	{
		std::string str = i;
		char separator = ' ';
		std::istringstream iss(str);
		std::string str_buf;
		
		int cnt{ 0 };
		string carNum{ "" };
		int minutes{ 0 };
		while (getline(iss, str_buf, separator)) {
			if (cnt == 0) {
				minutes += stoi(str_buf.substr(0, 2)) * 60 + stoi(str_buf.substr(3, 2));
			}
			else if(cnt == 1) {
				carNum = str_buf;
			}
			else if(cnt == 2) {
				if (str_buf == "IN")
				{
					m[carNum] = minutes;
				}
				else
				{
					if (minutes - m[carNum] < fees[0])
					{
						answer.emplace_back(fees[1]);
					}
					else
					{
						int costmin = (minutes - m[carNum]) / fees[2];
						int cost = costmin * fees[3];
						answer.emplace_back(cost);
					}
				}
			}
			cnt++;
			//std::cout << str_buf << std::endl;
		}
	}
	
	
	return answer;
}

int main()
{
	solution({ 180, 5000, 10, 600 }, 
		{ "05:34 5961 IN", 
		"06:00 0000 IN", 
		"06:34 0000 OUT", 
		"07:59 5961 OUT", 
		"07:59 0148 IN", 
		"18:59 0000 IN", 
		"19:09 0148 OUT", 
		"22:59 5961 IN", 
		"23:00 5961 OUT" });
	return 0;
}
