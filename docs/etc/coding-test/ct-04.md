```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

int main()
{
    vector<set<int>> vs;

    for (int i = 0; i < 10; i++)
    {
        set<int> s;
        s.insert(i);
        vs.push_back(s);
    }

    int insertNum{ 3 };
    int eraseNum{ 4 };

    for (auto item = vs.begin(); item != vs.end(); item++)
    {
        // insert num(3)
        item->insert(insertNum);
        
        // erase num(4)
        item->erase(eraseNum);
    }


    // print
    for (auto item = vs.begin(); item != vs.end(); item++)
    {
        for (auto sitem = item->begin(); sitem != item->end(); sitem++)
        {
            cout << *sitem << " ";
        }
        cout << endl;
    }

    return 0;
}
```
