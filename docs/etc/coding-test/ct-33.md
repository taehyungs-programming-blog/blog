#include <string>
#include <vector>
#include<sstream>

using namespace std;

#include <map>
#include <set>

vector<int> solution(vector<string> id_list, vector<string> report, int k) {

	map<string, set<string>> m;
	map<string, set<string>> banCnt;
	//map<string, int> m;

	for (auto i : report)
	{
		std::string str = i;
		char separator = ' ';
		std::istringstream iss(str);
		std::string str_buf;

		int cnt{ 0 };
		string user;
		string banUser;
		while (getline(iss, str_buf, separator)) {
			if (cnt == 0) {
				user = str_buf;
			}
			else if (cnt == 1) {
				banUser = str_buf;
				m[user].insert(banUser);
				banCnt[banUser].insert(user);
			}
			cnt++;

			//std::cout << str_buf << std::endl;
		}
	}

	vector<int> answer;
	for (auto i : id_list)
	{
		int ccc{ 0 };
		if (m.find(i) != m.end())
		{
			for (auto j : m[i])
			{
				if (banCnt.find(j) != banCnt.end())
				{
					if (banCnt[j].size() >= k) ccc++;
				}
			}
		}
		answer.emplace_back(ccc);
		
	}

	return answer;
}

}
