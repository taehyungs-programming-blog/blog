#include <string>
#include <vector>
#include <sstream>
#include <algorithm>

using namespace std;

vector<int> split(string input, char delimiter) {
    vector<int> answer;
    stringstream ss(input);
    string temp;
 
    while (getline(ss, temp, delimiter)) {
        answer.push_back(stoi(temp));
    }
 
    return answer;
}

string solution(string s) {

    int max = -2147483647 - 1;
    int min = 2147483647;

    auto v = split(s, ' ');
    
    sort(v.begin(), v.end());

    stringstream ss;
    ss << v.front() << " " << v.back();

    return ss.str();
}
