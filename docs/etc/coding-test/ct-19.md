#include <string>
#include <vector>
#include <stack>

using namespace std;

bool check(string s)
{
    if(s.empty()) return false;
    
    if(s[0] == ')' || 
      s[0] == ']' || 
      s[0] == '}')
    {
        return false;
    }
    
    stack<char> st;
    st.push(s[0]);
    
    for(int i = 1; i < s.length(); i++)
    {
        if( (st.top() == '(' && s[i] == ')') ||
            (st.top() == '{' && s[i] == '}') ||
            (st.top() == '[' && s[i] == ']') )
        {
            st.pop();
        }
        else
        {
            st.push(s[i]);
        }
    }
    
    return st.empty();
}

int solution(string s) {
    if(s.empty()) return 0;
    
    int answer = 0;
    for(int i = 0; i < s.length(); i++)
    {
        if(check(s)) answer++;
        char ch = s.front();
        s.erase(s.begin());
        s.push_back(ch);
        //s = s.substr(1, s.length() - 1) + s[0] ; // error
    }
    
    
    return answer;
}
