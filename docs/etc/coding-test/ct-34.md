#include <string>
#include <vector>

using namespace std;

#include<sstream>
#include <map>

struct Date
{
	int year;
	int month;
	int day;
};

Date parseDate(string s)
{
	std::string str = s;
	char separator = '.';
	std::istringstream iss(str);
	std::string str_buf;

	Date t;
	int cnt{ 0 };
	while (getline(iss, str_buf, separator)) {
		if (cnt == 0)
		{
			t.year = stoi(str_buf);
		}
		else if (cnt == 1)
		{
			t.month = stoi(str_buf);
		}
		else if (cnt == 2)
		{
			t.day = stoi(str_buf);
		}
		cnt++;
	}

	return t;
}

vector<int> solution(string today, vector<string> terms, vector<string> privacies) {
	
	Date _today = parseDate(today);
	
	map<int, pair< Date, string>> m;
	map<string, int> _term;

	for (auto i : terms)
	{
		std::string str = i;
		char separator = ' ';
		std::istringstream iss(str);
		std::string str_buf;

		string a;
		int b;
		int cnt{ 0 };
		while (getline(iss, str_buf, separator)) {
			if (cnt == 0)
			{
				a = str_buf;
			}
			else if (cnt == 1)
			{
				b = stoi(str_buf);
			}
			cnt++;
		}

		_term[a] = b;
	}

	int index{ 0 };
	for (auto i : privacies)
	{
		Date _day = parseDate(i.substr(0, i.size()-2));
		m[index] = pair<Date, string>(_day, i.substr(i.size() - 1, 1));
		index++;
	}

	vector<int> answer;

	for (int i = 0; i < m.size(); i++)
	{
		int termMonth = _term[m[i].second];

		int y{ 1 };


        // ì—¬ê¸° ë‚ ì§œê³„ì‚°ì´ ì–´ë ¤ì› ìŒ.
		if (termMonth >= 12)
		{
			y = termMonth / 12;
			termMonth = termMonth - y * 12;
			m[i].first.year += y;
		}

		m[i].first.month += termMonth;
		if (m[i].first.month > 12) {
			m[i].first.year += 1;
			m[i].first.month %= 13;
			m[i].first.month += 1;
		}

		m[i].first.day -= 1;
		if (m[i].first.day == 0) {
			m[i].first.day = 28;
			m[i].first.month -= 1;
			if (m[i].first.month == 0) {
				m[i].first.month = 12;
				m[i].first.year -= 1;
			}
		}

		// compare
		if (_today.year > m[i].first.year) {
			answer.emplace_back(i + 1);
			continue;
		}

		if (_today.year == m[i].first.year && _today.month > m[i].first.month) {
			answer.emplace_back(i + 1);
			continue;
		}

		if (_today.year == m[i].first.year && _today.month == m[i].first.month && _today.day > m[i].first.day) {
			answer.emplace_back(i + 1);
			continue;
		}
	}

	return answer;
}
