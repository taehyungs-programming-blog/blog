```sql
USE BaseballData

-- playerID : 선수 ID
-- yearID : 시즌 년도
-- teamID : 팀 명칭(약어)
-- G_batting : 출전 경기 타석
-- AB : 타수
-- H : 안타
-- R : 출루
-- 2B :2루타


-- Q1. 보스턴 소속 선수들의 정보만 출력
-- Q2. 보스턴 소속 선수들의 수(중복 제거)
-- Q3. 보스턴 팀이 2004년에 친 홈런 개수
-- Q4. 보스턴 팀 소속의 단일 년도 최다 홈런 사람

SELECT *
FROM batting
```

```sql
-- Q1. 보스턴 소속 선수들의 정보만 출력
SELECT *
FROM batting
WHERE teamId = 'BOS'
```

```sql
-- Q2. 보스턴 소속 선수들의 수(중복 제거)
SELECT COUNT(DISTINCT playerID)
FROM batting
WHERE teamId = 'BOS'
```

```sql
-- Q3. 보스턴 팀이 2004년에 친 홈런 개수
SELECT SUM(HR)
FROM batting
WHERE teamId = 'BOS' AND yearID = 2004
```

```sql
-- Q4. 보스턴 팀 소속의 단일 년도 최다 홈런 사람
SELECT TOP 1 *
FROM batting
WHERE teamId = 'BOS'
ORDER BY HR DESC
```
