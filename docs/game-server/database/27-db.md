```sql
USE BaseballData

-- Merge Join
SELECT *
FROM players AS p
    INNER JOIN salaries AS s
    ON p.playerID = s.playerID;
```

```sql
-- 중첩 루프(Nested Loop)
SELECT TOP 5 *
FROM players AS p
    INNER JOIN salaries AS s
    ON p.playerID = s.playerID;
```

-- Hash Join
