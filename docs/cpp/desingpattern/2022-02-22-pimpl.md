```cpp
// Point.h
class PointImpl;

class Point
{

    Point(int a= 0, int b = 0);
    void Print() const;

private:
    PointImpl* pImpl;
};
```

```cpp
// Point.cpp

class PointImpl
{
    int x, y, z;
};

Point::Point(int a, int b)
{
    pImpl = new PointImpl(a, b);
}

void Point::Print() const
{
    pImpl->Print();
}
```
