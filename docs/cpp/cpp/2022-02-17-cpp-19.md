* perfect forwarding이 그래서 뭔가?
    * 매개변수로 들어온 데이터(r, l value 모두)를 아무런 가공없이 다른 함수에 넘기고(forward) 싶다

😺 결론부터 보자면

```cpp
// 이렇게 해결가능
template<class F, class ... T>
decltype(auto) chronometry(F f, T&& ... arg)
{
	return f(std::forward<T>(arg)... );
}
```

* 그 넘기는 방법이 아래서 설명된다.

---

```cpp
#include <iostream>

void foo(int  n) {} 
void goo(int& r) { r = 20;} 

// 함수의 성능(시간)을 측정하는 템플릿을 만든다고 가정해 보자.
template<class F, class T>
void chronometry(F f, T arg)
{
	// 현재 시간 보관
	f(arg);
	// 함수 소요 시간 출력
}

int main()
{
	int n = 10;

    // 아래처럼 foo, goo함수에 인자를 보내고 싶은데 r, l value 어떤것이든 들어갈 수 있다.
    // 이렇게 어떤인자는 완벽하게 보내는 기법을 perfect forwarding이라 한다.
	chronometry(foo, 10); // foo(10)
	chronometry(goo, n);  // goo(n)

	std::cout << n << std::endl;
}
```

```cpp
// 일단 perfect forwarding을 구현하려하면 가장 쉽게 하는 생각은
// const T&로 받는것이다.
    // 이렇게 하면 r, l value 모두 받으니 해결됐겠지?
    // Nope! r, l value를 받기는 했으나 f에 넘길수가 없다.
    // goo의 경우 매개변수가 int& r인데 const int&로 변환시 int&로 변환이 불가하다
template<class F, class T>
void chronometry(F f,const T& arg)
{
	f(arg);
}
```

```cpp
void foo(int  n) {} 
void goo(int& r) { r = 20;} 

void hoo(int&& r) {}

// 우선 설명을 좀 쉽게 하기위해서 T = int로 고정하고
// r, l value를 받을수 있게 두 개로 만들면 어떨까?
template<class F>
void chronometry(F f, int& arg)
{
	f(arg);
}
template<class F>
void chronometry(F f, int&& arg)
{
	f(arg);
}

int main()
{
	hoo(10); // ok
	chronometry(hoo, 10);   // Error - 왜그런지는 아래서 설명

	int n = 10;
	chronometry(foo, 10); 
	chronometry(goo, n);  
	std::cout << n << std::endl;
}
```

// chronometry(hoo, 10); 호출시 10은 r-value이기에

void chronometry(F f, int&& arg)    // 여기로 호출되며

```cpp
// 해결책?

template<class F>
void chronometry(F f, int&& arg)
{
    // r-value로 들어왔던애는 다시 r-value로 캐스팅해주자
	f(static_cast<int&&(arg));
}
```

* 결론
    * perfect forwarding을 위해서는
    * r-value 함수, l-value 함수를 모두 제공해야한다.
    * 단, r-value 함수의 경우 들어온 매개변수를 다시 r-value로 캐스팅 해줘야 한다.

```cpp
#include <iostream>

void goo(int&  n) { n = 20;}
void hoo(int&& r) {}

/*
    // 매번 이렇게 코드를 짜야할까?

    template<class F>
    void chronometry(F f, int& arg)
    {
    //	f(arg);
        f( static_cast<int&>(arg) );
    }
    template<class F>
    void chronometry(F f, int&& arg)
    {
        f(static_cast<int&&>(arg));
    }
*/

template<class F, class T>
void chronometry(F f, T&& arg)
{
// forwarding reference를 이용하여 
    // & && -> &
    // && & -> &
    // && && -> &&
    // 일테니 해결가능?
//	f(static_cast<T&&>(arg));
// static_cast<T&&>도 길다 싶어서 std에서 forward를 제공한다.

	f( std::forward<T>(arg));
}

int main()
{	
	int n = 0;
	chronometry(goo, n);
	chronometry(hoo, 10);

	std::cout << n << std::endl; // 20
}
```

😺 그런데 매개변수가 하나만 전달될까?

```cpp
#include <iostream>

void foo() {}

int& goo(int a, int& b, int&& c)
{
	b = 20;
	return b;
}

// 가변인자 템플릿으로 간단히 해결가능.
template<class F, class ... T>
decltype(auto) chronometry(F f, T&& ... arg)
{
    // 리턴형까지 처리를 위해서 decltype(auto)를 넣음
        // 그냥 auto로 안쓴이유는 auto는 reference 특성을 버리게 된다.
	return f(std::forward<T>(arg)... );
}

int main()
{	
	int n = 0;

	int& ret = chronometry(goo, 10, n, 10);
				// goo(10, n, 10)
	chronometry(foo);
	ret = 100;
	
	std::cout << n << std::endl; 
}
```

😺 조금만 더 개선해보자.

```cpp
// 멤버함수까지 처리가능하게 해보자

#include <iostream>
#include <functional>

void foo(int n) {}

class Test
{
public:
	void f1(int n) { std::cout << "Test f1" << std::endl; }
};

template<class F, class ... T>
decltype(auto) chronometry(F f, T&& ... arg)
{
//	return f(std::forward<T>(arg)... );
// 멤버함수 포인터를 호출하려면 (obj.*f)(arg); 라는 규칙을 지켜줘야한다.
// 단, C++17 이후 invoke의 등장으로 아래와 같이 처리가능

	return std::invoke( f, std::forward<T>(arg)... );
}

int main()
{	
	chronometry(foo, 10);

	Test obj;
	chronometry(&Test::f1, &obj, 10);

}
```

```cpp
// 함수 객체까지 지원하게 해보자.
    // 함수 객체 -> () 연산자를 재정의해서 함수처럼 사용가능한 객체

#include <iostream>
#include <functional>

struct Functor
{
	void operator()(int n) &
	{
		std::cout << "operator() &" << std::endl;
	}
	void operator()(int n) &&
	{
		std::cout << "operator() &&" << std::endl;
	}
};

template<class F, class ... T>
decltype(auto) chronometry(F&& f, T&& ... arg)
{
    // std::forward<F>(f)를 하는이유는 f에 들어오면서 이름이 생겨
        // r-value가 l-value가 되어버림 따라서 std::forward<F>(f)를 써줘야한다.
	return std::invoke( std::forward<F>(f), 
				        std::forward<T>(arg)... );
}

int main()
{	
	Functor f;
	f(10); // f.operator()(10)
	Functor()(10); // operator() &&
	chronometry(f, 10);
	chronometry(Functor(), 10);
}
```
