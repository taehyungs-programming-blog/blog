```py
# models.py

from sqlalchemy import Column, Integer, String, Text, ForeignKey
from sqlalchemy.orm import relationship
from sqlalchemy.ext.declarative import declarative_base

Base = declarative_base()


class User(Base):
    __tablename__ = 'users'
    id = Column(Integer, primary_key=True, index=True)
    name = Column(String(255), index=True)
    email = Column(String(255), unique=True, index=True)
    posts = relationship('Post', back_populates='author')


class Post(Base):
    __tablename__ = 'posts'
    id = Column(Integer, primary_key=True, index=True)
    title = Column(String(255), index=True)
    content = Column(Text)
    user_id = Column(Integer, ForeignKey('users.id'))

    author = relationship('User', back_populates='posts')

```

```py
# data_operations.py

from sqlalchemy.orm import Session
from models import User, Post
from database import SessionLocal

def add_user(name: str, email: str) -> User:
    db = SessionLocal()
    db_user = User(name=name, email=email)
    db.add(db_user)
    db.commit()
    db.refresh(db_user)
    db.close()
    return db_user

def add_post(title: str, content: str, user_id: int) -> Post:
    db = SessionLocal()
    db_post = Post(title=title, content=content, user_id=user_id)
    db.add(db_post)
    db.commit()
    db.refresh(db_post)
    db.close()
    return db_post

def get_all_users() -> list:
    db = SessionLocal()
    users = db.query(User).all()
    db.close()
    return users

def get_user_posts(user_id: int) -> list:
    db = SessionLocal()
    posts = db.query(Post).filter(Post.user_id == user_id).all()
    db.close()
    return posts

def delete_post(post_id: int):
    db = SessionLocal()
    post_to_delete = db.query(Post).filter(Post.id == post_id).first()
    if post_to_delete:
        db.delete(post_to_delete)
        db.commit()
    db.close()

```
